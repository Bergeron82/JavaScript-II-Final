1 | Manipulate the DOM

This function was created to show interactivity with the DOM. It will use JavaScript to modify an HTML element. An event listener on the button will run the function when the button is clicked and change the text color and font weight of this paragraph.

function modifyText() {
    let paragraph = document.getElementById('modify_html');

}

2 | Add and Remove HTML Nodes

The next two functions will use createElement() and removeChild() methods to add a new paragraph node and remove the new node using event listeners on the respective buttons.

function addNode() {
    let add = document.createElement('p');
    let newPara = document.createTextNode(`"It is, in fact, nothing short of a miracle that the modern methods of instruction have not yet entirely strangled the holy curiosity of inquiry." - Albert Einstein`);

    add.setAttribute('style', 'font-weight:700; font-size:1.4rem; border:1px solid gray; padding: 3rem');
    add.appendChild(newPara);
    let placement = document.getElementById('node');
    placement.appendChild(add);
}
const addBtn = document.getElementById('add_node');
addBtn.addEventListener('click', addNode, false);
function removeNode() {
    let remove = document.getElementById('node').firstChild;
    let container = remove.parentNode;
    container.removeChild(remove);
}
const removeBtn = document.getElementById('remove_node');
removeBtn.addEventListener('click', removeNode, false);

The following function will display a variety of information related to the Browser Object Model via a click event listener on the "Show Info" button.

function showBom() {
    let info = `<h3>Window Location</h3>${window.location.href}<br>`;

    info += `<h3>Window Size</h3>width: ${window.innerWidth}px<br>`;

const bomBtn = document.getElementById('bom_btn');

5 | Create Scripts Using Built-in JS Classes

The following function will use the new Date() method to show a running clock of the current local time and the current date. This function was put inside another function called "run" so that the setInterval() method used to keep the clock running would only run when the button is clicked.

function run() {
    function showDateInfo() {
      let today = new Date();
      let showDate = document.getElementById('date');
      showDate.setAttribute('style', 'background-color:#2e2e2e;');
      showDate.innerHTML = `Today's Date: ${today.toLocaleDateString()}<br>
       Current Time: ${today.toLocaleTimeString()}`;
  }
  setInterval(showDateInfo, 1);
}
const dateBtn = document.getElementById('date_btn');
dateBtn.addEventListener('click', run, false);

7 | Validate JS and Apply Debugging Techniques

The following function uses the built-in JavaScript error object with the try and catch statements to validate input errors. The last few lines of code in the function use an if statement to run a message letting the user know they have succeeded with their input.

function handleErrors() {
    let errorMessage = document.getElementById('hint');
    let userInput = document.getElementById('user_input').value;
    try {
      if (userInput == '') throw 'full of emptiness';
      if (userInput < 100) throw 'less than one benjamin';
      if (userInput > 150) throw 'more than two benjamins';
      if (isNaN(userInput)) throw 'text';
    }
    catch(err) {
      errorMessage.textContent = `User input cannot be ${err}.`;
    }
    if (userInput > 100 && userInput < 150 && userInput !=='' && !isNaN(userInput)) {
      errorMessage.textContent="You've found the sweet spot!" ;
      errorMessage.setAttribute('style', 'font-size:1.5rem; color:#ff7676;' );
    }
   }
const check = document.getElementById('check_code');
check.addEventListener('click', handleErrors, false);

8 | Apply Techniques for Security Issues

The following function uses a variable set to a string of special characters. To prevent bad actors from entering code into the text box a for loop and if/else statement is run to evaluate the user input. If the user information does not include any of the specified special characters, they are then allowed to submit the form. Press the Show Form button to view the demonstration!

function validate() {
    let comments = document.getElementById('comments');
    let specialChars = '<>/$&{}][}`%';

    for (let i = 0; i < comments.value.length; i++) {
      if (specialChars.indexOf(comments.value.charAt(i)) !=-1) {
          alert('Evil characters detected, try again!');
          comments.focus();
      } else {
          alert('Congrats, your dream vacation will be submitted!');
      } break;
    }
}
const submit=document.getElementById('submit');
submit.addEventListener('click', validate, false);
